    <script>
        // YAHAN APNI API KEY LIKHEIN:
        const API_KEY = "apni api key yaha likhe";
        
        let currentLang = 'english';
        let fetchedData = null;

        lucide.createIcons();

        function setLang(lang) {
            currentLang = lang;
            const buttons = ['english', 'romanUrdu', 'urduScript'];
            buttons.forEach(b => {
                const btn = document.getElementById(`btn-${b}`);
                if(b === lang) {
                    btn.className = "bg-white shadow text-blue-600 px-4 py-1.5 rounded-md text-sm font-medium transition-colors";
                } else {
                    btn.className = "text-slate-500 hover:text-slate-700 px-4 py-1.5 rounded-md text-sm font-medium transition-colors";
                }
            });
            updateUI();
        }

        async function searchMedicine() {
            const query = document.getElementById('searchInput').value;
            if (!query.trim()) return;

            document.getElementById('searchIcon').classList.add('hidden');
            document.getElementById('loadingIcon').classList.remove('hidden');
            document.getElementById('searchBtn').disabled = true;
            document.getElementById('btnText').innerText = "Searching...";
            
            document.getElementById('errorBox').classList.add('hidden');
            document.getElementById('notFoundBox').classList.add('hidden');
            document.getElementById('resultsSection').classList.add('hidden');
            fetchedData = null;

            const systemPrompt = `You are a highly knowledgeable pharmacist. Provide accurate info about the medicine the user searches. You MUST return ONLY a valid JSON object with this exact structure: 
            { "isMedicine": true, "medicineName": "...", "english": { "uses": "...", "dosage": "...", "sideEffects": "...", "precautions": "...", "disclaimer": "..." }, "romanUrdu": { "uses": "...", "dosage": "...", "sideEffects": "...", "precautions": "...", "disclaimer": "..." }, "urduScript": { "uses": "...", "dosage": "...", "sideEffects": "...", "precautions": "...", "disclaimer": "..." } }. 
            If the input is not a medicine, set isMedicine to false and return this: { "isMedicine": false, "errorMsg_EN": "Sorry, not found.", "errorMsg_RU": "Maaf kijiye, nahi mila.", "errorMsg_UR": "معاف کیجئے، نہیں ملا۔" }`;

            try {
                // Is line mein maine latest API version (v1) set kar diya hai
                const response = await fetch(`https://generativelanguage.googleapis.com/v1/models/gemini-1.5-flash:generateContent?key=${API_KEY}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: `${systemPrompt}\n\nUser Input: ${query}` }] }]
                    })
                });

                const result = await response.json();
                
                if (result.error) {
                    throw new Error(result.error.message);
                }

                let aiResponse = result.candidates[0].content.parts[0].text;
                aiResponse = aiResponse.replace(/```json/g, '').replace(/```/g, '').trim();
                fetchedData = JSON.parse(aiResponse);
                updateUI();
                
            } catch (err) {
                const errorBox = document.getElementById('errorBox');
                errorBox.innerText = 'Asal Masla: ' + err.message;
                errorBox.classList.remove('hidden');
            } finally {
                document.getElementById('searchIcon').classList.remove('hidden');
                document.getElementById('loadingIcon').classList.add('hidden');
                document.getElementById('searchBtn').disabled = false;
                document.getElementById('btnText').innerText = "Search";
            }
        }

        function updateUI() {
            if (!fetchedData) return;

            if (fetchedData.isMedicine === false) {
                const notFoundBox = document.getElementById('notFoundBox');
                const notFoundText = document.getElementById('notFoundText');
                
                if (currentLang === 'english') notFoundText.innerText = fetchedData.errorMsg_EN;
                else if (currentLang === 'romanUrdu') notFoundText.innerText = fetchedData.errorMsg_RU;
                else if (currentLang === 'urduScript') notFoundText.innerText = fetchedData.errorMsg_UR;
                
                notFoundBox.classList.remove('hidden');
                notFoundBox.classList.add('flex');
                document.getElementById('resultsSection').classList.add('hidden');
                return;
            }

            document.getElementById('notFoundBox').classList.add('hidden');
            document.getElementById('notFoundBox').classList.remove('flex');
            const resultsSection = document.getElementById('resultsSection');
            resultsSection.classList.remove('hidden');

            if (currentLang === 'urduScript') {
                resultsSection.setAttribute('dir', 'rtl');
            } else {
                resultsSection.setAttribute('dir', 'ltr');
            }

            const langData = fetchedData[currentLang];
            
            document.getElementById('medName').innerText = fetchedData.medicineName;
            document.getElementById('medUses').innerText = langData.uses;
            document.getElementById('medDosage').innerText = langData.dosage;
            document.getElementById('medSideEffects').innerText = langData.sideEffects;
            document.getElementById('medPrecautions').innerText = langData.precautions;
            document.getElementById('medDisclaimer').innerText = langData.disclaimer;
        }
    </script>
